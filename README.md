## Installation

`yarn add nyse-holidays`

## Usage

```ts
import { isHoliday, getHolidays } from "nyse-holidays";

console.log(isHoliday(new Date("2021-01-01"))); // true

console.log(isHoliday(new Date("2022-01-01"))); // false

console.log(getHolidays(2021));
```
